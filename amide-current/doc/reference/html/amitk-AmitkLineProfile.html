AmitkLineProfile

typedef struct {
  GObject parent;

  AmitkView view;
  amide_real_t angle; /* in radians */
  gboolean visible;

  /* in base coordinate frame.  These are updated by the canvas */
  AmitkPoint start_point;
  AmitkPoint end_point;
} AmitkLineProfile;

typedef struct {
